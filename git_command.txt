git 命令：
#查看工作区状态
git status
#修改
git add filename
#提交
git commit -m '修改说明'
#查看修改日志
git log
git log --pretty=oneline
#版本回退 HEAD^^：表示回退上上一个版本  HEAD~100 表示回退100个版本
git reset --hard HEAD^
#查看文件内容
cat filename
#回复到任意版本
git reset --hard 'commint_id'
#操作命令记录查看（附带commit_id）
git reflog
#查看工作区和版本库中的差异
git diff HEAD -- filename
#恢复工作区文件到修改之前
git checkout -- filename
#恢复暂存区的文件到工作区
git reset HEAD filename
#删除文件
rm filename
#删除版本库中的文件,并且需要提交
git rm filename
